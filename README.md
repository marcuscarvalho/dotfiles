```json
    // When using ZSH, Agnoster and Meslo LG M for Powerline fonts

    "editor.rulers": [ 100 ],
    "editor.detectIndentation": true,
    "editor.lineHeight": 24,
    "workbench.colorCustomizations": {
        "editorLineNumber.activeForeground": "#FFF"
    }
}
```
